\def\makeName#1{

\def\infoItem#1#2{\hbox to 0.4\hsize{ 
		\hbox to 0.12\hsize{\it #1: \hfil} #2\hfil}}
\def\myLine#1{\hbox to #1\hsize{\vbox to 2mm{\hrule width #1\hsize}}}
\def\fatLine#1{\hbox to #1\hsize{\vbox to 2mm{\sclr \hrule width #1\hsize depth 1mm \mclr}}}

\def\secFrac{0.23}
\def\restFrac{0.75}
\def\startSec#1{\hbox{\fatLine{\secFrac} \secfont \sclr #1 \mclr} \smallskip}
\def\finishSec{\bigskip}

\def\onerow#1#2#3{\hbox to\hsize{%
\vbox{\noindent\hsize=\secFrac\hsize{\bf #1 -- #2 \par \noindent \dashfill \hfill}}%
\vbox{\noindent \kern2mm \hsize=\restFrac\hsize #3}
}\bigskip}
\def\onerowNoDue#1#2{\hbox to\hsize{%
\vbox{\noindent\hsize=\secFrac\hsize{\bf #1}}%
\vbox{\noindent \kern2mm \hsize=\restFrac\hsize #2}
}\smallskip}

\def\secFracShort{0.12}
\def\restFracShort{0.85}
\def\startSecShort#1{\hbox{\fatLine{\secFracShort} \secfont \sclr #1 \mclr} \smallskip}
\def\onerowshort#1#2{\hbox to\hsize{%
\vbox{\noindent\hsize=\secFracShort\hsize{\bf #1 }}%
\vbox{\noindent \kern2mm \hsize=\restFracShort\hsize #2}
}\smallskip}

\def\twocols#1#2{\hbox to\hsize{%
\vbox{\noindent\hsize=0.5\hsize #1}%
\vbox{\noindent\hsize=0.5\hsize #2}%
}}
